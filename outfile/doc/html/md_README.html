Usage

import com.seevi.SVCameraDevice.SVCameraDevice;
import com.seevi.SVCameraDevice.SVCameraDeviceInfo;

QuickStart

//枚举出所有SVCameraDevice设备
List<SVCameraDeviceInfo> lst = SVCameraDevice.enumDevice(this);
//取出想要操作的SVCameraDevice设备
SVCameraDeviceInfo device = lst.get(0);
if(device == null) {
Log.e(TAG, "no device found!");
return;
}
boolean bRet;
//创建一个SVCameraDeviceController实例.
SVCameraDevice controller = new SVCameraDevice(this);
//打开指定SVCameraDevice设备.
bRet = controller.openDevice(device);
//取得版本号.
String strModel = controller.getCameraModel();
Log.e(TAG, strModel);
String strSWVersion = controller.getSoftwareVersion();
Log.e(TAG, strSWVersion);
//ZOOM
bRet = controller.Zoom(SVCameraDevice.ZOOM_TYPE.ZOOM_TYPE_IN, 2);
//Move.
bRet = controller.Move(SVCameraDevice.DIRECTION_TYPE.DIRECTION_RIGHT, 2);
